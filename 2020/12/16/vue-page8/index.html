Vue学习笔记8——编写音乐播放器应用

音乐播放器的功能

  1. 歌曲搜索
  2. 歌曲播放
  3. 歌曲方面
  4. 歌曲评论
  5. 专辑动画
  6. 播放MV

逻辑

  1. 歌曲搜索
  • 按下回车：v-on绑定keyup.enter
  • 查询数据：axios接口 v-model绑定表单和数据
  • 渲染数据：v-for 数组 that保存this的值
  1. 歌曲播放
  • 点击播放：v-on 绑定click
  • 歌曲地址获取：axios接口 歌曲id
  • 歌曲地址设置：v-bind 绑定播放器audio的播放音乐地址src
  1. 歌曲封面
  • 点击播放：同上，增加逻辑
  • 歌曲封面获取：axios接口 歌曲id
  • 歌曲封面设置：v-bind 绑定图片的src
  1. 歌曲评论
  • 点击播放：同上，增加逻辑
  • 歌曲评论获取：axios接口 歌曲id
  • 歌曲评论渲染：v-for渲染到页面列表dl
  1. 专辑动画
  • 监听音乐播放：v-on绑定audio的@play事件 用play方法播放
  • 监听音乐暂停：v-on绑定audio的@pause事件 用pause方法暂停
  • 操纵类名：v-bind绑定专辑图片的class 增加类名isPlaying 然后css在增加的类名上面增加动画旋转效果
  1. 播放MV
  • MV图标显示：v-if/v-show
  • MV地址获取：axios接口 获取mvid
  • 遮罩层：（遮盖播放器界面，显示MV播放界面）v-show v-on
  • MV地址设置：v-bind

代码如下，由于篇幅原因，CSS代码省略

<div class="wrap">
    <!-- 播放器主体区域 -->
    <div class="play_wrap" id="player">
        <div class="search_bar">
            <input autocomplete="off" v-model="query"
            @keyup.enter="searchMusic"/>
        </div>
        <div class="center_on">
            <!-- 歌曲名字列表 -->
            <div class="song_wrapper">
                <ul class="song_list">
                    <li v-for="item in musicList" style="list-style: none;">
                        <a href="javascript:;" @click="playMusic(item.id)">
                            <img  src="../img/1.gif">
                        </a>
                        <b>{{item.name}}</b>
                        <span v-if="item.mvid!=0" @click="playMV(item.mvid)"><img src="../img/4.png"></span>
                    </li>
                </ul>
            </div>
            <!-- 歌曲专辑封面 -->
            <div class="player_con" :class="{playing:isPlaying}">
                <img v-bind:src="musicCover" class="">
            </div>
            <!-- 评论容器 -->
            <div class="comment_wrapper">
                <h5 class="title">热门留言</h5>
                <div class='comment_list'>
                    <dl v-for="item in hotComments">
                        <dt><img :src="item.user.avatarUrl"></dt>
                        <dd class="name">{{item.nickname}}</dd>
                        <dd>{{item.content}}</dd>
                    </dl>
                </div>
            </div>
        </div>
        <!-- 音乐播放栏 -->
        <div class="audio_con">
            <audio ref='audio' @play="play" @pause="pause" v-bind:src="musicUrl" controls autoplay loop class="myaudio"></audio>
        </div>
        <!-- MV视频播放 -->
        <div class="video_con" v-show="isShow" >
            <video :src="mvUrl" controls="controls"></video>
            <div class="mask" @click="hide"></div>
        </div>
    </div>
</div>
/*
    1.歌曲搜索接口
        请求地址:https://autumnfish.cn/search
        请求方法:get
        请求参数：keywords（查询关键字）
        响应内容：歌曲搜索结果
    2.歌曲url获取接口
        请求地址:https:/autumnfish.cn/song/url
        请求方法:get
        请求参数:id(查询关键字)
        响应内容:歌曲url地址
    3.歌曲详情获取
        请求地址:https://autumnfish.cn/song/detail
        请求方法:get
        请求参数:ids（歌曲id）
        响应内容:歌曲详情(包括封面信息)
    4.热门评论获取
        请求地址:https://autumnfish.cn/comment/hot?type=0
        请求方法:get
        请求参数:id(歌曲id,地址中的type固定为0)
        响应内容:歌曲的热门评论
*/

var app=new Vue({
    el:"#player",
    data:{
        //查询关键字
        query:"",
        //歌曲数组
        musicList:[],
        //歌曲地址
        musicUrl:"",
        //歌曲封面
        musicCover:"",
        //歌曲评论
        hotComments:[],
        //动画播放状态
        isPlaying:false,
        //遮罩层的显示状态
        isShow:false,
        //mv地址
        mvUrl:"",
    },
    methods:{
        //歌曲搜索
        searchMusic:function(){
            var that=this;
            axios.get("https://autumnfish.cn/search?keywords="+this.query)
            .then(function(response){
                //console.log(response);
                that.musicList=response.data.result.songs;
                console.log(response.data.result.songs);
            },function(err){});
        },
        //歌曲播放
        playMusic:function(musicId){
            var that=this;
            console.log(musicId);
            //获取歌曲地址
            axios.get("https:/autumnfish.cn/song/url?id="+musicId)
            .then(function(response){
                // console.log(response);
                //console.log(response.data.data[0].url);
                that.musicUrl=response.data.data[0].url;
            },function(err){ })

            //歌曲详情获取
            axios.get("https://autumnfish.cn/song/detail?ids="+musicId)
            .then(function(response){
                // console.log(response);
                console.log(response.data.songs[0].al.picUrl);
                that.musicCover=response.data.songs[0].al.picUrl;
            },function(err){})

            //歌曲评论获取
            axios.get("https://autumnfish.cn/comment/hot?type=0&id="+musicId)
            .then(function(response){
                // console.log(response);
                console.log(response.data.hotComments);
                that.hotComments=response.data.hotComments;
            },function(err){})
        },
        play:function(){
            // console.log("play");
            this.isPlaying=true;
        },
        pause:function(){
            // console.log("pause");
            this.isPlaying=false;
        },
        //播放mv
        playMV:function(mvid){
            var that=this;
            axios.get("https://autumnfish.cn/mv/url?id="+mvid)
            .then(function(response){
                console.log(response.data.data.url);
                that.isShow=true;
                console.log(that.isShow);
                that.mvUrl=response.data.data.url;
                console.log(that.mvUrl);
            },function(err){  })
        },
        //隐藏mv
        hide:function(){
            this.isShow=false;
        }
    }
})

总结

  1. 不同接口的需求数据不同，所以要仔细阅读文档数据
  2. 页面结构复杂的情况下，通过debug审查元素去定位相关元素，快速定位
  3. 响应式的数据都需要在data中定义
